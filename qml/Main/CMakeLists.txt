set(MODULE_NAME Main)
message(STATUS "Configuring: ${MODULE_NAME}")

file(GLOB_RECURSE QML_FILE_LIST
    RELATIVE "${CMAKE_CURRENT_SOURCE_DIR}"
    CONFIGURE_DEPENDS
    "${CMAKE_CURRENT_SOURCE_DIR}/*.qml"
)

qt_add_qml_module(
    ${MODULE_NAME}
    URI "${MODULE_NAME}"
    VERSION 1.0
    QML_FILES ${QML_FILE_LIST}
    RESOURCE_PREFIX /
)

source_group("${MODULE_NAME}" FILES ${QML_FILE_LIST})

list(APPEND QML_IMPORT_PATH "${CMAKE_CURRENT_SOURCE_DIR}")
set(QML_IMPORT_PATH "${QML_IMPORT_PATH}" PARENT_SCOPE)
